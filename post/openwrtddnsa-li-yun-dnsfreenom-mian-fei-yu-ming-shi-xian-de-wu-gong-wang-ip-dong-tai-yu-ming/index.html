config ddns 'global'
        option ddns_dateformat '%F %R'
        option ddns_loglines '250'
        option upd_privateip '0'

config service 'WuHanXz'
        option service_name 'aliyun.com'
        option enabled '1'
        option lookup_host 'example.com'

EOF
